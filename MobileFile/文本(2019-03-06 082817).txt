public static void main(String[] args) throws UnknownHostException,			IOException {		// 以下一行代码就连接上了ip地址，并且找到监听8888端口port的网络服务器		Socket socket = new Socket("localhost", 8888);				//只要连接上服务器，就启动一条单独的线程接收服务器发过来的信息		new ClientThread5(socket).start();				PrintStream ps = new PrintStream(socket.getOutputStream());		String line = null;		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));		// 线程阻塞		while ((line = br.readLine()) != null) {			ps.println(line);			ps.flush();		}							}}class ClientThread5 extends Thread {	private Socket socket;	public ClientThread5(Socket socket) {		this.socket = socket;	}	@Override	public void run() {		String line = null;		BufferedReader br;		try {			br = new BufferedReader(new InputStreamReader(					socket.getInputStream()));						while ((line = br.readLine()) != null) {				// 客户端读取服务器写过来的信息				System.out.println(line);			}					} catch (IOException e) {			e.printStackTrace();		}	}
